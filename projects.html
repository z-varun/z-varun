def validate_security_policy(resource):
    """
    Validates cloud resources against
    security policies using OPA
    """
    policy_result = opa.evaluate(
        policy="security/baseline",
        input_data=resource
    )
    
    if not policy_result.allowed:
        raise SecurityViolation(
            policy_result.violations
        )
    
    return policy_result
